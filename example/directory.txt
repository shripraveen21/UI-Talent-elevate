    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const tabs = {
                filters: document.getElementById('tab-filters'),
                employees: document.getElementById('tab-employees'),
            };
            const panels = {
                filters: document.getElementById('panel-filters'),
                employees: document.getElementById('panel-employees'),
            };

            const setActiveTab = (tabName) => {
                // Deactivate all tabs
                Object.values(tabs).forEach(tab => {
                    tab.classList.remove('tab-active');
                    tab.classList.add('text-gray-500', 'hover:text-gray-700');
                });
                Object.values(panels).forEach(panel => panel.classList.add('hidden'));

                // Activate the selected one
                tabs[tabName].classList.add('tab-active');
                tabs[tabName].classList.remove('text-gray-500', 'hover:text-gray-700');
                panels[tabName].classList.remove('hidden');
            };
            
            // Set initial active tab
            setActiveTab('filters');

            tabs.filters.addEventListener('click', () => setActiveTab('filters'));
            tabs.employees.addEventListener('click', () => setActiveTab('employees'));
        });
    </script>
